= 5/9 . (59 - 32)

= 5/9 . 27 = 15°C

= (59 - 32) / (9.5) + 273

= 27/45 + 273 = 33,066°K

°R = 4/5.(°K - 273)

= 4/5 . (288 - 273)

= 4/5 . 15 = 12°R

°C = °K - 273

= 288 - 273 = 15°C

°F = 9/5.(°K - 273) + 32

= 9/5 . (288 - 273) + 32

= 9/5 . 15 + 32

= 27 + 32 = 59°F
